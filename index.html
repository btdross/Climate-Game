    let currentRound = 0;

    let score = 0;


    // Get DOM elements

    const gameContainer = document.querySelector('.game-container');

    const scoreValue = document.getElementById('score-value');

    const roundNumber = document.getElementById('round-number');

    const scenarioText = document.getElementById('scenario-text');

    const optionsContainer = document.getElementById('options-container');

    const outcomeState = document.getElementById('outcome-state');

    const outcomeText = document.getElementById('outcome-text');

    const nextBtn = document.getElementById('next-btn');

    const finalScreen = document.getElementById('final-screen');

    const finalMessage = document.getElementById('final-message');

    const finalScoreText = document.getElementById('final-score-text');

    const restartBtn = document.getElementById('restart-btn');

    const gameState = document.getElementById('game-state');

    const illustrationContainer = document.getElementById('illustration-container');


    // Function to start or restart the game

    function startGame() {

        currentRound = 0;

        score = 0;

        scoreValue.textContent = score;

        finalScreen.style.display = 'none';

        gameContainer.classList.remove('hidden');

        gameState.classList.remove('hidden');

        outcomeState.classList.add('hidden');

        displayRound();

    }


    // Function to display the current round's content

    function displayRound() {

        if (currentRound >= gameData.length) {

            endGame();

            return;

        }


        const roundData = gameData[currentRound];

        roundNumber.textContent = roundData.round;

        scenarioText.innerHTML = `<h3>${roundData.title}</h3><br>${roundData.scenario}`;

        illustrationContainer.innerHTML = roundData.illustration;

        optionsContainer.innerHTML = '';

        

        roundData.options.forEach((option, index) => {

            const button = document.createElement('button');

            button.className = 'option-btn';

            button.textContent = option.text;

            button.onclick = () => handleChoice(index);

            optionsContainer.appendChild(button);

        });

    }


    // Function to handle a user's choice

    function handleChoice(choiceIndex) {

        const chosenOption = gameData[currentRound].options[choiceIndex];

        

        // Update score

        score += chosenOption.points;

        scoreValue.textContent = score;


        // Display outcome and next round button

        gameState.classList.add('hidden');

        outcomeState.classList.remove('hidden');

        

        const outcomeHtml = `

            <div class="report-heading">Your Choice's Impact:</div>

            <p>${chosenOption.impact}</p>

            <br>

            <div class="report-heading">Pros:</div>

            <p>${chosenOption.pros}</p>

            <br>

            <div class="report-heading">Cons:</div>

            <p>${chosenOption.cons}</p>

        `;

        outcomeText.innerHTML = outcomeHtml;

        

        // Change score color based on impact

        if (chosenOption.points > 0) {

            scoreValue.style.color = 'var(--primary-color)';

        } else {

            scoreValue.style.color = 'var(--negative-color)';

        }

    }


    // Function to advance to the next round

    function nextRound() {

        currentRound++;

        // Reset score color

        scoreValue.style.color = 'var(--primary-color)';

        gameState.classList.remove('hidden');

        outcomeState.classList.add('hidden');

        displayRound();

    }


    // Function to end the game and show the final score

    function endGame() {

        gameContainer.classList.add('hidden');

        finalScreen.style.display = 'flex';

        finalScoreText.textContent = `Final Score: ${score}`;

        

        if (score >= 100) {

            finalMessage.textContent = "A Sustainable Future Achieved!";

        } else if (score >= 50) {

            finalMessage.textContent = "A Challenging but Hopeful Future.";

        } else {

            finalMessage.textContent = "The Future is in Peril.";

        }

    }


    // Event listeners

    nextBtn.addEventListener('click', nextRound);

    restartBtn.addEventListener('click', startGame);


    // Initial call to start the game

    startGame();
